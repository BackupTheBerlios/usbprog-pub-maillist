This patch fixes following linker error:

gcc -g -o usbprog main.o -lusb -I ../lib/  -lavrupdate -pg
/usr/lib64/gcc/x86_64-suse-linux/4.2.1/../../../../x86_64-suse-linux/bin/ld: cannot find -lavrupdate
collect2: ld returned 1 exit status


Signed-off-by: Bernhard Walle <bernhard.walle at gmx.de>

---
 usbprog-old/Makefile |    2 +-
 1 file changed, 1 insertion(+), 1 deletion(-)

--- a/usbprog-old/Makefile
+++ b/usbprog-old/Makefile
@@ -1,6 +1,6 @@
 all:
 	gcc -g -c main.c -I ../lib/ -pg
-	gcc -g -o usbprog main.o -lusb -I ../lib/  -lavrupdate -pg
+	gcc -g -o usbprog main.o -lusb -L ../lib/  -lavrupdate -pg
 
 install:
 	cp usbprog /usr/bin
